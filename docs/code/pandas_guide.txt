Pandas for Data Science - Complete Reference
=============================================

Loading Data:
    import pandas as pd

    # Load CSV
    df = pd.read_csv('data.csv')

    # Load with options
    df = pd.read_csv('data.csv', parse_dates=['date_col'], index_col='id')

    # Load Excel
    df = pd.read_excel('data.xlsx', sheet_name='Sheet1')

    # Load JSON
    df = pd.read_json('data.json')

    # Load from SQL
    import sqlite3
    conn = sqlite3.connect('database.db')
    df = pd.read_sql('SELECT * FROM customers WHERE active = 1', conn)

Exploring Data:
    df.head(10)             # First 10 rows
    df.tail()               # Last 5 rows
    df.shape                # (rows, columns) as tuple
    df.info()               # Column names, types, null counts
    df.describe()           # Count, mean, std, min, max, quartiles
    df.dtypes               # Data type of each column
    df.columns.tolist()     # List of column names
    df.isnull().sum()       # Count of missing values per column
    df.nunique()            # Count of unique values per column
    df.value_counts()       # Frequency of each value in a Series

Selecting and Filtering:
    df['column']                        # Single column as Series
    df[['col1', 'col2']]               # Multiple columns as DataFrame
    df.iloc[0]                         # First row by integer position
    df.iloc[0:5, 1:3]                  # Rows 0-4, columns 1-2
    df.loc[0, 'column']                # Row by label and column name
    df.loc[df['age'] > 30]             # Filter rows by condition
    df[(df['age'] > 30) & (df['city'] == 'NYC')]  # Multiple conditions
    df.query('age > 30 and salary > 50000')        # SQL-like syntax

Cleaning Data:
    df.dropna()                         # Drop rows with any null
    df.dropna(subset=['age', 'salary']) # Drop only if these cols are null
    df.fillna(0)                        # Replace all nulls with 0
    df.fillna({'age': df['age'].mean(), 'city': 'Unknown'})  # Per column
    df.drop_duplicates()                # Remove exact duplicate rows
    df.drop_duplicates(subset=['email'])# Remove by specific column
    df['name'] = df['name'].str.strip() # Remove leading/trailing whitespace
    df['email'] = df['email'].str.lower()  # Lowercase
    df['phone'] = df['phone'].str.replace('-', '')  # Remove characters

Feature Engineering:
    # Mathematical transformations
    df['age_squared'] = df['age'] ** 2
    df['log_income'] = np.log1p(df['income'])  # log(1+x) handles zeros
    df['income_per_age'] = df['income'] / df['age']

    # Binary flags
    df['is_senior'] = (df['age'] >= 65).astype(int)
    df['high_value'] = (df['spend'] > df['spend'].quantile(0.9)).astype(int)

    # Date features
    df['date'] = pd.to_datetime(df['date'])
    df['year'] = df['date'].dt.year
    df['month'] = df['date'].dt.month
    df['day_of_week'] = df['date'].dt.dayofweek  # 0=Monday
    df['is_weekend'] = df['day_of_week'].isin([5, 6]).astype(int)
    df['quarter'] = df['date'].dt.quarter

    # Rolling and cumulative stats
    df['rolling_7d_avg'] = df['sales'].rolling(window=7).mean()
    df['rolling_30d_max'] = df['sales'].rolling(window=30).max()
    df['cumulative_sales'] = df['sales'].cumsum()

    # Lag features (previous values)
    df['sales_lag_1'] = df['sales'].shift(1)   # Previous day
    df['sales_lag_7'] = df['sales'].shift(7)   # Same day last week

Grouping and Aggregation:
    # Basic groupby
    df.groupby('category')['sales'].sum()
    df.groupby('category')['sales'].agg(['sum', 'mean', 'count'])

    # Multiple columns and multiple aggregations
    df.groupby(['region', 'product']).agg({
        'sales': ['sum', 'mean'],
        'quantity': 'sum',
        'returns': 'count'
    })

    # Pivot table
    df.pivot_table(
        values='sales',
        index='region',
        columns='product',
        aggfunc='sum',
        fill_value=0
    )

    # Apply custom function
    df.groupby('category')['sales'].apply(lambda x: x.quantile(0.9))

Merging DataFrames:
    # Inner join (only matching rows)
    merged = pd.merge(df1, df2, on='customer_id', how='inner')

    # Left join (all rows from df1)
    merged = pd.merge(df1, df2, on='customer_id', how='left')

    # Concatenate rows
    combined = pd.concat([df1, df2], ignore_index=True)

Saving Data:
    df.to_csv('output.csv', index=False)
    df.to_excel('output.xlsx', index=False, sheet_name='Results')
